class Solution {
    public boolean searchMatrix(int[][] matrix, int target) {
        if(matrix == null || matrix.length == 0)
            return false;
        if(matrix[0].length == 0)
            return false;
        for(int[] arr : matrix)
        {
            int left = 0; 
            int right = arr.length;
            if(arr[left] > target)
                continue;
            while(left < right)
            {
                int mid = (left + right)/2;
                if(arr[mid] == target)
                {
                    return true;
                }
                else if(arr[mid] > target)
                {
                    right = mid;
                }
                else if(arr[mid] < target)
                {
                    left = mid+1;
                }
            }
        }
        return false;
    }
}
