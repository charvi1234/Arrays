class Solution {
    public int maxProfit(int[] prices) {
        int max = 0;
        int i = 0;
        for(int j = 1; j < prices.length;)
        {
            if(prices[j] - prices[i] > 0)
            {
                if(prices[j] - prices[i] > max)
                {
                    max = prices[j] - prices[i];
                }
                j++;
            }
            else
            {
                i = j;
                j = i+1;
            }
        }
        return max;
    }
}
